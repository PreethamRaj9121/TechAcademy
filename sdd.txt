#!/bin/bash

# Rename all files to lower case
for file in *
do
    if [ -f "$file" ]; then
        newname=$(echo "$file" | tr '[:upper:]' '[:lower:]')
        mv "$file" "$newname"
    fi
done

# Rename all directories to upper case
for dir in */
do
    if [ -d "$dir" ]; then
        newname=$(echo "$dir" | tr '[:lower:]' '[:upper:]')
        mv "$dir" "$newname"
    fi
done
