
#!/bin/bash

# Ask user for the file name to rename
echo "Enter the name of the file to rename: "
read file

# Check if the given path is a regular file
if [ -f "$file" ]; then
    # Rename the file to lower case
    newname=$(echo "$file" | tr '[:upper:]' '[:lower:]')
    mv "$file" "$newname"
    echo "File renamed to $newname"
else
    echo "Error: $file is not a regular file."
fi
